# Standalone pipeline to validate a Function App after deploy.
# Runs on a self-hosted agent that uses `az login --identity`.

trigger:
 - none

parameters:
- name: Environment
  displayName: "Environment"
  type: string
  default: Nprod
  values:
    - Nprod
    - Prod

- name: subscriptionId
  displayName: "Target Azure subscription ID"
  type: string

- name: resourceGroup
  displayName: "Resource group of the Function App"
  type: string

- name: functionAppName
  displayName: "Function App name"
  type: string

- name: expectedPython
  displayName: "Expected Python runtime (must match deploy)"
  type: string
  default: "3.10"
  values:
    - "3.9"
    - "3.10"
    - "3.11"

- name: expectedFunctionCount
  displayName: "Expected indexed function count"
  type: number
  default: 16

pool:
  name: 'WindowsDevOps'

steps:
- checkout: self
  displayName: 'Checkout repo'

- task: PowerShell@2
  displayName: 'Validate ${{ parameters.functionAppName }}'
  inputs:
    targetType: 'inline'
    script: |
      $ErrorActionPreference = 'Stop'
      az login --identity --allow-no-subscriptions | Out-Null
      az account set --subscription "${{ parameters.subscriptionId }}"

      pwsh -File "$(System.DefaultWorkingDirectory)/Utilities/validate_deploy.ps1" `
        -ResourceGroup "${{ parameters.resourceGroup }}" `
        -FunctionApp "${{ parameters.functionAppName }}" `
        -ExpectedPython "${{ parameters.expectedPython }}" `
        -ExpectedFunctionCount ${{ parameters.expectedFunctionCount }}
