# validate_deploy.ps1
# Diagnoses a Python V2 Function App that deployed successfully but doesn't
# show its functions in the portal.
#
# Prereqs:
#   - Azure CLI installed and `az login` done (or running inside an
#     AzureCLI@2 task with an active service connection).
#   - Read access on the Function App + its resource group.
#
# Usage:
#   pwsh ./Utilities/validate_deploy.ps1 -ResourceGroup <rg> -FunctionApp <name>
#
# Optional:
#   -ExpectedPython "3.10"     # what your deploy pipeline pinned
#   -ExpectedFunctionCount 16  # how many functions you registered
#
# Exit codes:
#   0  all checks passed
#   1  one or more checks failed (details printed)

[CmdletBinding()]
param(
    [Parameter(Mandatory = $true)][string]$ResourceGroup,
    [Parameter(Mandatory = $true)][string]$FunctionApp,
    [string]$ExpectedPython = "3.10",
    [int]$ExpectedFunctionCount = 16
)

$ErrorActionPreference = 'Stop'
$failures = @()

function Write-Pass([string]$msg) { Write-Host "  PASS  $msg" -ForegroundColor Green }
function Write-Fail([string]$msg) { Write-Host "  FAIL  $msg" -ForegroundColor Red;   $script:failures += $msg }
function Write-Warn([string]$msg) { Write-Host "  WARN  $msg" -ForegroundColor Yellow }
function Write-Section([string]$msg) { Write-Host "`n=== $msg ===" -ForegroundColor Cyan }

# ---------------------------------------------------------------------------
Write-Section "1. App settings"

$settingsJson = az functionapp config appsettings list `
    --resource-group $ResourceGroup --name $FunctionApp -o json
if (-not $settingsJson) {
    Write-Fail "Could not read app settings. Check RG + FA name and your az login."
    exit 1
}
$settings = ($settingsJson | ConvertFrom-Json) | ForEach-Object { @{ name = $_.name; value = $_.value } }
$lookup = @{}
foreach ($s in $settings) { $lookup[$s.name] = $s.value }

function Require-Setting([string]$key, [string]$expected = $null, [string]$hint = "") {
    if (-not $lookup.ContainsKey($key)) {
        Write-Fail "$key is NOT set. $hint"
        return
    }
    $val = $lookup[$key]
    if ($expected -and $val -ne $expected) {
        Write-Fail "$key = '$val' (expected '$expected'). $hint"
        return
    }
    Write-Pass "$key = '$val'"
}

Require-Setting "FUNCTIONS_WORKER_RUNTIME"    "python"
Require-Setting "FUNCTIONS_EXTENSION_VERSION" "~4"
Require-Setting "WEBSITE_RUN_FROM_PACKAGE"    $null `
    "Should usually be '1' for zip deploy via AzureFunctionApp@1."

if (-not $lookup.ContainsKey("AzureWebJobsFeatureFlags") `
    -or ($lookup["AzureWebJobsFeatureFlags"] -notmatch "EnableWorkerIndexing")) {
    Write-Fail "AzureWebJobsFeatureFlags is missing 'EnableWorkerIndexing'. This is the #1 cause of zero functions in the portal. Fix: az functionapp config appsettings set --resource-group $ResourceGroup --name $FunctionApp --settings AzureWebJobsFeatureFlags=EnableWorkerIndexing"
} else {
    Write-Pass "AzureWebJobsFeatureFlags includes EnableWorkerIndexing"
}

$hasAwjStorage = $lookup.ContainsKey("AzureWebJobsStorage") `
    -or $lookup.ContainsKey("AzureWebJobsStorage__accountName")
if ($hasAwjStorage) {
    if ($lookup.ContainsKey("AzureWebJobsStorage__accountName")) {
        Write-Pass "AzureWebJobsStorage configured via identity (__accountName)"
        if (-not $lookup.ContainsKey("AzureWebJobsStorage__credential") `
            -or $lookup["AzureWebJobsStorage__credential"] -ne "managedidentity") {
            Write-Warn "AzureWebJobsStorage__credential is not 'managedidentity'. Required for MI-based bindings."
        }
    } else {
        Write-Warn "AzureWebJobsStorage is set as a connection string. Works, but consider switching to __accountName + __credential=managedidentity to retire the connection string."
    }
} else {
    Write-Fail "Neither AzureWebJobsStorage nor AzureWebJobsStorage__accountName is set. Queue triggers and host bookkeeping will fail."
}

# ---------------------------------------------------------------------------
Write-Section "2. Runtime (Python version)"

$linuxFx = az functionapp config show `
    --resource-group $ResourceGroup --name $FunctionApp `
    --query "linuxFxVersion" -o tsv
if (-not $linuxFx) {
    Write-Warn "linuxFxVersion is empty (Windows plan?). If Windows, set FUNCTIONS_WORKER_RUNTIME_VERSION instead."
} elseif ($linuxFx -eq "Python|$ExpectedPython") {
    Write-Pass "linuxFxVersion = '$linuxFx'"
} else {
    Write-Fail "linuxFxVersion = '$linuxFx' (expected 'Python|$ExpectedPython'). Mismatch causes native deps (cryptography, pandas) to fail to load and indexing dies silently."
}

# ---------------------------------------------------------------------------
Write-Section "3. Deployed functions"

$funcsJson = az functionapp function list `
    --resource-group $ResourceGroup --name $FunctionApp -o json 2>$null
if (-not $funcsJson) {
    Write-Fail "az functionapp function list returned nothing. Either the app has 0 indexed functions (indexing failed) or you don't have rights to list functions."
} else {
    $funcs = $funcsJson | ConvertFrom-Json
    $count = @($funcs).Count
    if ($count -eq 0) {
        Write-Fail "0 functions indexed. Check Log Stream for indexing errors and verify EnableWorkerIndexing is set."
    } elseif ($count -lt $ExpectedFunctionCount) {
        Write-Fail "$count of $ExpectedFunctionCount functions indexed. Some bp.py imports are failing. Check Log Stream."
        $names = @($funcs) | ForEach-Object { ($_.name -split "/")[-1] }
        Write-Host "  Indexed: $($names -join ', ')"
    } else {
        Write-Pass "$count functions indexed (expected $ExpectedFunctionCount)"
        $names = @($funcs) | ForEach-Object { ($_.name -split "/")[-1] }
        Write-Host "  Functions: $($names -join ', ')"
    }
}

# ---------------------------------------------------------------------------
Write-Section "4. Last 20 lines from Log Stream (indexing errors usually appear here)"

try {
    $logs = az webapp log download `
        --resource-group $ResourceGroup --name $FunctionApp `
        --log-file "$env:TEMP\$FunctionApp-logs.zip" 2>$null
    if (Test-Path "$env:TEMP\$FunctionApp-logs.zip") {
        Write-Host "  Downloaded logs to $env:TEMP\$FunctionApp-logs.zip"
        Write-Host "  Expand and inspect /LogFiles/Application for 'Failed to index' or 'ImportError'."
    } else {
        Write-Warn "Could not download logs. View Log Stream live in the portal: Function App -> Log stream"
    }
} catch {
    Write-Warn "Log download failed: $($_.Exception.Message). Use the portal's Log Stream instead."
}

# ---------------------------------------------------------------------------
Write-Section "Summary"

if ($failures.Count -eq 0) {
    Write-Host "All checks passed." -ForegroundColor Green
    exit 0
} else {
    Write-Host "$($failures.Count) check(s) failed:" -ForegroundColor Red
    foreach ($f in $failures) { Write-Host " - $f" -ForegroundColor Red }
    Write-Host "`nNext steps:"
    Write-Host " 1. Fix the failed settings (commands shown above)"
    Write-Host " 2. Restart the Function App: az functionapp restart -g $ResourceGroup -n $FunctionApp"
    Write-Host " 3. Wait ~60s, then re-run this script"
    exit 1
}
